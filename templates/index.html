Fallback: análise automática do erro return analisar_erro_automatico(erro_mensagem) except Exception as e: print(f"❌ Exception: {type(e).__name__}: {e}") return analisar_erro_automatico(erro_mensagem) def analisar_erro_automatico(erro_mensagem): """Análise automática quando Claude falha""" erro_lower = erro_mensagem.lower() #
